log_query = """SELECT
log.external_member_id AS external_member_id,
ees.oration_id AS oration_id,
ees.primary_oration_id AS primary_oration_id,
ees.org_name AS org_name,
log.datetime AS datetime,
IF(log.event_name CONTAINS 'member_invited',1,0) AS member_email,
IF(log.event_name CONTAINS 'dependent_invited',1,0) AS dependent_email,
IF(log.event_name CONTAINS 'member_visit',1,0) AS member_visit,
IF(log.event_name CONTAINS 'registration_complete',1,0) AS registration_complete,
IF(log.event_name CONTAINS 'order_submitted',1,0) AS order_submitted,
IF(log.event_name CONTAINS 'campaign_call',1,0) AS campaign_call
FROM
log.detail log LEFT JOIN
(SELECT
primary_oration_id AS primary_oration_id,
oration_id AS oration_id,
member_id AS external_member_id,
org_short_name AS org_name,
date_part AS yob,
cnt AS dependents,
relationship AS relationship
FROM
ees.enrollment
GROUP BY
primary_oration_id, oration_id, external_member_id, org_name, yob, dependents, relationship) ees
ON
log.external_member_id = ees.external_member_id
WHERE 
(event_name CONTAINS 'member_invited' OR
event_name CONTAINS 'dependent_invited' OR
event_name CONTAINS 'member_visit' OR
event_name CONTAINS 'registration_complete' OR
event_name CONTAINS 'order_submitted' OR
event_name CONTAINS 'campaign_call') 
AND
ees.org_name NOT IN ('ORG0006','ORG0007')
GROUP BY
external_member_id,  oration_id, primary_oration_id, org_name, datetime, 
member_email, dependent_email, member_visit, registration_complete, order_submitted, campaign_call;"""

optimization_query = """SELECT
b.oration_id AS oration_id,
b.primary_oration_id AS primary_oration_id,
b.org_short_name AS org_name,
b.date_part AS yob,
b.cnt AS rel_cnt,
b.relationship AS rel,
a.custom_name AS drug_name,
a.strength AS strength,
a.lf_chain AS lf_chain,
a.opt_chain AS opt_chain,
a.opt_quantity AS opt_quantity,
a.opt_days_supply AS opt_days_supply,
a.lf_days_supply AS lf_days_supply,
a.lf_fills_per_year AS lf_fills_per_year,
a.order_status AS order_status,
a.order_date AS order_date,
SUM(lf_member_cost) AS lf_member_cost,
SUM(lf_org_cost) AS lf_org_cost,
SUM(a.member_annual_cost) AS member_annual_cost,
SUM(a.member_annual_savings) AS member_annual_savings,
AVG(a.member_cost_per_fill) AS member_cost_per_fill,
AVG(a.member_savings_per_fill) AS member_savings_per_fill,
SUM(a.org_annual_cost) AS org_annual_cost,
SUM(a.org_annual_savings) AS org_annual_savings, 
AVG(a.org_cost_per_fill) AS org_cost_per_fill,
AVG(a.org_savings_per_fill) AS org_savings_per_fill,
SUM(a.rebate) AS rebate
FROM
opt.summary a LEFT JOIN ees.enrollment b
ON
a.oration_id = b.oration_id
AND 
a.primary_oration_id = b.primary_oration_id
AND
a.org_name = b.org_short_name
WHERE
b.org_short_name NOT IN ('ORG0006','ORG0007')
GROUP BY
oration_id, primary_oration_id, org_name, yob, rel_cnt, rel, drug_name, strength,
lf_chain, opt_chain, opt_quantity, opt_days_supply, lf_days_supply, lf_fills_per_year, order_status, order_date;"""

campaign_query = """SELECT * 
FROM

# Calculate current cost (last fill cost)
optimization['lf_member_cost'] = optimization['lf_fills_per_year'] * optimization['lf_member_cost']
optimization['lf_org_cost'] = optimization['lf_fills_per_year'] * optimization['lf_org_cost']

# Calculate proposed cost (optimized cost)

optimization['o_total_cost'] = optimization['o_member_cost'] + optimization['o_org_cost']

# Calculate savings (last fill - optimized cost)
optimization['member_annual_savings_calc'] = optimization['lf_member_cost'] - optimization['o_member_cost']
optimization['org_annual_savings_calc'] = optimization['lf_org_cost'] - optimization['o_org_cost']
optimization['total_annual_savings_calc'] = optimization['lf_total_cost'] - optimization['o_total_cost']

# Calculate difference between system reported and manually calculated savings

                            (optimization['org_annual_savings_calc'] - optimization['org_annual_savings']))

# Create meta-attributes columns for different registration types
attribution['registration_complete_calls'] = np.nan
attribution['registration_complete_emails'] = np.nan

# Create meta-attributes columns for different order types
attribution['order_submitted_calls'] = np.nan
attribution['order_submitted_emails'] = np.nan

# If registration = 0, then no attribution is necessary
attribution.loc[attribution['registration_complete'] == 0,'registration_complete_calls'] = 0
attribution.loc[attribution['registration_complete'] == 0,'registration_complete_emails'] = 0

attribution.loc[attribution['order_submitted'] == 0,'order_submitted_calls'] = 0
attribution.loc[attribution['order_submitted'] == 0,'order_submitted_emails'] = 0

########################################################################################
### Registration logic
# If they recieved an email, didn't visit the site but registered, then flag as call registration
reg_ix_1 = (((attribution['member_email'] > 0) | (attribution['dependent_email'] > 0))  & 
                (attribution['member_visit'] == 0) & 
                (attribution['registration_complete'] > 0))

attribution.loc[reg_ix_1,'registration_complete_calls'] = attribution.loc[reg_ix_1,'registration_complete']
attribution.loc[reg_ix_1,'registration_complete_emails'] = 0

# If they receievd an email, visited the site and registered and were never called, then flag as email registration
reg_ix_2 = (((attribution['member_email'] > 0) | (attribution['dependent_email'] > 0))  &

              (attribution['campaign_call'] == 0) &
